71template <uint8_t max>
72uint8_t CMD<max>::exe(char* input, uint8_t nullAsArg)
73{
74 char* nextCmd = nullptr;
75 char* nextArg = nullptr;
76 char* currCmd = nullptr;
77 char* currArg = nullptr;
78 int16_t cmdIdx = -1;
79 uint8_t argCnt = 0;
80 uint8_t cmdCnt = 0;
81
82 // Set pointer to first command
83 nextCmd = input;
84
85 // While next command is found
86 do
87 {
88 // Reset argument counter
89 argCnt = 0;
90
91 // Copy command start address
92 currCmd = nextCmd;
93
94 // Find next command
95 nextCmd = sStd::tok(nextCmd, delimiter[sCMD_CC]);
96
97 // Find start address of first argument
98 nextArg = sStd::tok(currCmd, delimiter[sCMD_CA]);
99
100 // If argument part is found
101 if (nextArg)
102 {
103 // While command to argument delimiter is found
104 do
105 {
106 // Stop parsing arguments if argument limit is reached
107 if (argCnt == maxArgs) break;
108
109 // Copy argument start address
110 currArg = nextArg;
111
112 // Find next argument
113 nextArg = sStd::tok(nextArg, delimiter[sCMD_AA]);
114
115 // If current argument is not \0 or if nullAsArg is set to 1
116 if (*currArg || nullAsArg)
117 {
118 // Copy argument's start address to argument array
119 args[argCnt] = currArg;
120
121 // Increase argument counter
122 argCnt++;
123 }
124 }
125 while (nextArg);
126 }
127
128 // If current command is not \0
129 if (*currCmd)
130 {
131 // Find command index in command list
132 cmdIdx = findCmd(currCmd, cmdList, cmdListLen);
133
134 // Call fallback handler if command was not found
135 if (cmdIdx == -1) cmdFallback((const char**)currCmd, 0);
136 else // Command is found, call command handler and pass arguments
137 {
138 cmdList[cmdIdx].cmdHandler((const char**)args, argCnt);
139
140 // Increas command counter
141 cmdCnt++;
142 }
143 }
144 }
145 while (nextCmd);
146
147 // Return number of executed commands
148 return cmdCnt;
149}

165static int16_t findCmd(const char* input, const CMDList* cmdList, const uint16_t len)
166{
167 for (uint16_t i = 0; i < len; i++)
168 {
169 // If command is found, return index
170 if (sStd::cmp(input, cmdList[i].cmd) == SSTD_OK) return i;
171 }
172
173 return -1;
174}
